36 void print(std::ostream& os) const {
37 if (_data->cols() == 1) {

50// Logger class
51/*
52------
53Notes:
54------
55- This class is meant to be a high speed signal logger with low computational
56impact on the application thread. The logger writes to file on a separate
57thread.
58- Variables to be logged are registered before the logger starts. The variables
59have to persist across the lifetime of the logger, otherwise a segfault will
60occur.
61- Eigen vector and matrices, double, int and bool are supported
62- Time since start is automatically logged as the first column of the output
63file.
64- Logging frequency is specified in Hz and is 100 Hz by default.
65- Output is a csv file with following format, and the variables will always be
66logged with the eigen values first, followed by double, int and bool values:
67
68time, Vector1_0, Vector1_1, Vector2_0, Vector2_1, double1, int1, bool1
690, 5.0, 5.0, 5.0, 5.0, 2.0, 23, 0
700.01, 5.0, 5.0, 5.0, 5.0, 2.0, 23, 0
710.02, 5.0, 5.0, 5.0, 5.0, 2.0, 23, 0
72...
73*/

135 bool addToLog(const Eigen::MatrixBase<Derived>& var,
136 const std::string& var_name = "") {
137 if (_f_is_logging) {
138 return false;
139 }
140 auto e = new EigenVector<Derived>(&var);
141 _eigen_vars_to_log.push_back(dynamic_cast<IEigenVector*>(e));
142 _num_eigen_vars_to_log += var.size();
143 for (uint i = 0; i < var.size(); i++) {
144 if (!var_name.empty()) {
145 _eigen_header += var_name + "__" + std::to_string(i) + ", ";
146 } else {
147 _eigen_header += "var" +
148 std::to_string(_eigen_vars_to_log.size()) +
149 "__" + std::to_string(i) + ", ";
150 }
151 }
152 return true;
153 }
