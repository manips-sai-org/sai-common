435 template <typename Derived>
436 std::string RedisClient::encodeEigenMatrix(
437  const Eigen::MatrixBase<Derived>& matrix) {
438  std::string s = "[";
439  if (matrix.cols() == 1) { // Column vector
440  // [[1],[2],[3],[4]] => "[1,2,3,4]"
441  for (int i = 0; i < matrix.rows(); ++i) {
442  if (i > 0) s.append(",");
443  s.append(std::to_string(matrix(i, 0)));
444  }
445  } else { // Matrix
446  // [[1,2,3,4]] => "[1,2,3,4]"
447  // [[1,2],[3,4]] => "[[1,2],[3,4]]"
448  for (int i = 0; i < matrix.rows(); ++i) {
449  if (i > 0) s.append(",");
450  // Nest arrays only if there are multiple rows
451  if (matrix.rows() > 1) s.append("[");
452  for (int j = 0; j < matrix.cols(); ++j) {
453  if (j > 0) s.append(",");
454  s.append(std::to_string(matrix(i, j)));
455  }
456  // Nest arrays only if there are multiple rows
457  if (matrix.rows() > 1) s.append("]");
458  }
459  }
460  s.append("]");
461  return s;
462 }
