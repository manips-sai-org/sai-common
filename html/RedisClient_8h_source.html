438 template <typename Derived>
439 std::string RedisClient::encodeEigenMatrix(
440  const Eigen::MatrixBase<Derived>& matrix) {
441  std::string s = "[";
442  if (matrix.cols() == 1) { // Column vector
443  // [[1],[2],[3],[4]] => "[1,2,3,4]"
444  for (int i = 0; i < matrix.rows(); ++i) {
445  if (i > 0) s.append(",");
446  s.append(std::to_string(matrix(i, 0)));
447  }
448  } else { // Matrix
449  // [[1,2,3,4]] => "[1,2,3,4]"
450  // [[1,2],[3,4]] => "[[1,2],[3,4]]"
451  for (int i = 0; i < matrix.rows(); ++i) {
452  if (i > 0) s.append(",");
453  // Nest arrays only if there are multiple rows
454  if (matrix.rows() > 1) s.append("[");
455  for (int j = 0; j < matrix.cols(); ++j) {
456  if (j > 0) s.append(",");
457  s.append(std::to_string(matrix(i, j)));
458  }
459  // Nest arrays only if there are multiple rows
460  if (matrix.rows() > 1) s.append("]");
461  }
462  }
463  s.append("]");
464  return s;
465 }
